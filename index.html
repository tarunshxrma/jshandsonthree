Question 1 : Find the number of digits

Answer:

const Find_Digits = (N) => 
{
  let count=0; 
  while(N!==0){
    count++;
    N=Math.floor(N/10);
  }
  return count;
};

Question 2 : Find the Fives

Answer:

const Find_Five = (N) => 
{
  let count=0;
  while(N!==0){
    let lastdigit=N%10;
    if(lastdigit==5){
      count++;
    }
    N=Math.floor(N/10);
  }
  return count;
};

Question 3 : Find Sum

Answer:

const findSum = (n) => {
  let it,ans=0;
  for(it=1;it<=n;it++){
    if(it%2===0){
      ans+=it;
    }
  }
  return ans;
};

Question 4 : Find the sum of the digits of a given number.

Answer:

const Number_Sum = (N) => 
{
	let sum=0;
	while(N!==0){
	  let digit=N%10;
	  sum=sum+digit;
	  N=Math.floor(N/10);
	}
	return sum;
};

Question 6 : Print the Pattern.

Answer:

const Print_pattern = (N) => 
{
  for(let r=1;r<=N;r++){
    s="";
    for(let c=1;c<=r;c++){

    }
    console.log(s);
  }
};

{
